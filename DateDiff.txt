SELECT DATEDIFF(Day,[Check-In Date],[Check-Out Date]) From [HOTEL_MANAGEMENT].[dbo].[ROOM_RESERVATION]; 
UPDATE [HOTEL_MANAGEMENT].[dbo].[TOTALPAYMENT]
SET [Days of Stay]=DATEDIFF(Day,[Check-In Date],[Check-Out Date]) 
From [HOTEL_MANAGEMENT].[dbo].[ROOM_RESERVATION]
WHERE [HOTEL_MANAGEMENT].[dbo].[TOTALPAYMENT].[Room No]=[HOTEL_MANAGEMENT].[dbo].[ROOM_RESERVATION].[Room No]
